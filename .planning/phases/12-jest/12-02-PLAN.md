**Important: execFile callback mocking (Pitfall 4)**
gitContext.ts uses `promisify(execFile)`. The mock must call the callback parameter.
Since execFile has multiple overloads, and `promisify` wraps the callback-based version, mock implementations must handle the arguments correctly:

```typescript
// Helper to create a mock that returns specific stdout for all git commands
function mockAllCommands(outputs: Record<string, string>) {
  mockedExecFile.mockImplementation((_cmd: any, args: any, _opts: any, callback?: any) => {
    const cb = typeof _opts === 'function' ? _opts : callback
    // Determine which command based on args
    const argsStr = Array.isArray(args) ? args.join(' ') : ''
    let stdout = ''
    if (argsStr.includes('log') && argsStr.includes('--oneline')) stdout = outputs.commits_today ?? ''
    else if (argsStr.includes('diff') && argsStr.includes('--stat')) stdout = outputs.diff_lines ?? ''
    else if (argsStr.includes('log') && argsStr.includes('--format=%ai')) stdout = outputs.first_commit_time ?? ''
    else if (argsStr.includes('rev-parse')) stdout = outputs.repo_path ?? ''
    if (cb) cb(null, stdout, '')
    return {} as any
  })
}
```
